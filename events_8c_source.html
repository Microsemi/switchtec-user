173 int switchtec_event_summary_set(struct switchtec_event_summary *sum,
174  enum switchtec_event_id e,
175  int index)
176 {
177  uint64_t bit = events[e].summary_bit;
178 
179  switch (events[e].type) {
180  case GLOBAL:
181  sum->global &= bit;
182  break;
183  case PART:
184  if (index == SWITCHTEC_EVT_IDX_LOCAL) {
185  sum->local_part |= bit;
186  } else if (index == SWITCHTEC_EVT_IDX_ALL) {
187  set_all_parts(sum, bit);
188  } else if (index < 0 || index >= ARRAY_SIZE(sum->part)) {
189  errno = EINVAL;
190  return -EINVAL;
191  } else {
192  sum->part[index] |= bit;
193  }
194  break;
195  case PFF:
196  if (index == SWITCHTEC_EVT_IDX_ALL) {
197  set_all_pffs(sum, bit);
198  } else if (index < 0 || index >= ARRAY_SIZE(sum->pff)) {
199  errno = EINVAL;
200  return -EINVAL;
201  } else {
202  sum->pff[index] |= bit;
203  }
204  break;
205  }
206 
207  return 0;
208 }

246 int switchtec_event_summary_iter(struct switchtec_event_summary *sum,
247  enum switchtec_event_id *e,
248  int *idx)
249 {
250  int bit;
251 
252  if (!idx || !e)
253  return -EINVAL;
254 
255  *idx = 0;
256 
257  bit = ffs(sum->global) - 1;
258  if (bit >= 0) {
259  *e = global_event_bits[bit];
260  sum->global &= ~(1 << bit);
261  return 1;
262  }
263 
264  for (*idx = 0; *idx < ARRAY_SIZE(sum->part); (*idx)++) {
265  bit = ffs(sum->part[*idx]) - 1;
266  if (bit < 0)
267  continue;
268 
269  *e = part_event_bits[bit];
270  sum->part[*idx] &= ~(1 << bit);
271  return 1;
272  }
273 
274  for (*idx = 0; *idx < ARRAY_SIZE(sum->pff); (*idx)++) {
275  bit = ffs(sum->pff[*idx]) - 1;
276  if (bit < 0)
277  continue;
278 
279  *e = pff_event_bits[bit];
280  sum->pff[*idx] &= ~(1 << bit);
281  return 1;
282  }
283 
284  return 0;
285 }

302

304  return -EINVAL;
305 
306  if (!res)
307  res = &res_tmp;
308 
309  ret = switchtec_event_summary(dev, res);
310  if (ret)
311  return ret;
312 
313  if (chk->global & res->global)
314  return 1;
315

317  return 1;
318

320  if (chk->part[i] & res->part[i])
321  return 1;
322 
323  for (i = 0; i < SWITCHTEC_MAX_PORTS; i++)
324  if (chk->pff[i] & res->pff[i])
325  return 1;
326 
327  return 0;

361 int switchtec_event_wait_for(struct switchtec_dev *dev,
362  enum switchtec_event_id e, int index,
363  struct switchtec_event_summary *res,
364  int timeout_ms)
365 {
366  struct timeval tv;
367  long long start, now;
368  struct switchtec_event_summary wait_for = {0};
369  int ret;
370 
371  ret = switchtec_event_summary_set(&wait_for, e, index);
372  if (ret)
373  return ret;
374 
375  ret = switchtec_event_ctl(dev, e, index,
376  SWITCHTEC_EVT_FLAG_CLEAR |
377  SWITCHTEC_EVT_FLAG_EN_POLL,
378  NULL);
379  if (ret)
380  return ret;
381 
382  ret = gettimeofday(&tv, NULL);
383  if (ret)
384  return ret;
385 
386  now = start = ((tv.tv_sec) * 1000 + tv.tv_usec / 1000);
387 
388  while (1) {
389  ret = switchtec_event_wait(dev, timeout_ms > 0 ?
390  now - start + timeout_ms : -1);
391  if (ret < 0)
392  return ret;
393 
394  if (ret == 0)
395  goto next;
396 
397  ret = switchtec_event_check(dev, &wait_for, res);
398  if (ret < 0)
399  return ret;
400 
401  if (ret)
402  return 1;
403 
404 next:
405  ret = gettimeofday(&tv, NULL);
406  if (ret)
407  return ret;
408 
409  now = ((tv.tv_sec) * 1000 + tv.tv_usec / 1000);
410 
411  if (timeout_ms > 0 && now - start >= timeout_ms)
412  return 0;
413  }
414 }
