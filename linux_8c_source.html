197  int device, int func)
198 {
199  char path[PATH_MAX];

201  struct dirent *dirent;
202  DIR *dir;
203 
204  snprintf(path, sizeof(path),
205  "/sys/bus/pci/devices/%04x:%02x:%02x.%x/switchtec",
206  domain, bus, device, func);
207 
208  dir = opendir(path);
209  if (!dir)
210  goto err_out;
211 
212  while ((dirent = readdir(dir))) {
213  if (dirent->d_name[0] != '.')
214  break;
215  }
216 
217  if (!dirent)
218  goto err_close;
219 
220  /*
221  * Should only be one switchtec device, if there are
222  * more then something is wrong
223  */
224  if (readdir(dir))
225  goto err_close;
226 
227  snprintf(path, sizeof(path), "/dev/%s", dirent->d_name);
228  printf("%s\n", path);
229  dev = switchtec_open(path);
230 
231  closedir(dir);
232  return dev;
233 
234 err_close:
235  closedir(dir);
236 err_out:
237  errno = ENODEV;

292 int switchtec_list(struct switchtec_device_info **devlist)
293 {
294  struct dirent **devices;
295  int i, n;
296  char link_path[PATH_MAX];
297  char pci_path[PATH_MAX];
298  struct switchtec_device_info *dl;
299 
300  n = scandir(sys_path, &devices, scan_dev_filter, alphasort);
301  if (n <= 0)
302  return n;
303 
304  dl = *devlist = calloc(n, sizeof(struct switchtec_device_info));
305 
306  if (!dl) {
307  for (i = 0; i < n; i++)
308  free(devices[n]);
309  free(devices);
310  errno = ENOMEM;
311  return -errno;
312  }
313 
314  for (i = 0; i < n; i++) {
315  snprintf(dl[i].name, sizeof(dl[i].name),
316  "%s", devices[i]->d_name);
317  snprintf(dl[i].path, sizeof(dl[i].path),
318  "/dev/%s", devices[i]->d_name);
319 
320  snprintf(link_path, sizeof(link_path), "%s/%s/device",
321  sys_path, devices[i]->d_name);
322 
323  if (readlink(link_path, pci_path, sizeof(pci_path)) > 0)
324  snprintf(dl[i].pci_dev, sizeof(dl[i].pci_dev),
325  "%s", basename(pci_path));
326  else
327  snprintf(dl[i].pci_dev, sizeof(dl[i].pci_dev),
328  "unknown pci device");
329 
330  snprintf(link_path, sizeof(link_path), "%s/%s",
331  sys_path, devices[i]->d_name);
332 
333  get_device_str(link_path, "product_id", dl[i].product_id,
334  sizeof(dl[i].product_id));
335  get_device_str(link_path, "product_revision",
336  dl[i].product_rev, sizeof(dl[i].product_rev));
337  get_fw_version(link_path, dl[i].fw_version,
338  sizeof(dl[i].fw_version));
339 
340  free(devices[n]);
341  }
342 
343  free(devices);
344  return n;
345 }

368 static int submit_cmd(struct switchtec_linux *ldev, uint32_t cmd,
369  const void *payload, size_t payload_len)
370 {
371  int ret;
372  size_t bufsize = payload_len + sizeof(cmd);
373  char buf[bufsize];
374 
375  cmd = htole32(cmd);
376  memcpy(buf, &cmd, sizeof(cmd));
377  memcpy(&buf[sizeof(cmd)], payload, payload_len);
378 
379  ret = write(ldev->fd, buf, bufsize);
380 
381  if (ret < 0)
382  return ret;
383 
384  if (ret != bufsize) {
385  errno = EIO;
386  return -errno;
387  }
388 
389  return 0;
390 }

392 static int read_resp(struct switchtec_linux *ldev, void *resp,
393  size_t resp_len)
394 {
395  int32_t ret;
396  size_t bufsize = sizeof(uint32_t) + resp_len;
397  char buf[bufsize];
398 
399  ret = read(ldev->fd, buf, bufsize);
400 
401  if (ret < 0)
402  return ret;
403 
404  if (ret != bufsize) {
405  errno = EIO;
406  return -errno;
407  }
408 
409  memcpy(&ret, buf, sizeof(ret));
410  if (ret)
411  errno = ret;
412 
413  if (!resp)
414  return ret;
415 
416  memcpy(resp, &buf[sizeof(ret)], resp_len);
417 
418  return ret;
419 }

442 static int get_class_devices(const char *searchpath,
443  struct switchtec_status *status)
444 {
445  int i;
446  ssize_t len;
447  char syspath[PATH_MAX];
448  glob_t paths;
449  int found = 0;
450  const size_t MAX_LEN = 256;
451 
452  snprintf(syspath, sizeof(syspath), "%s*/*/device", searchpath);
453  glob(syspath, 0, NULL, &paths);
454 
455  for (i = 0; i < paths.gl_pathc; i++) {
456  char *p = paths.gl_pathv[i];
457 
458  len = readlink(p, syspath, sizeof(syspath));
459  if (len <= 0)
460  continue;
461 
462  p = dirname(p);
463 
464  if (!status->class_devices) {
465  status->class_devices = calloc(MAX_LEN, 1);
466  strcpy(status->class_devices, basename(p));
467  } else {
468  len = strlen(status->class_devices);
469  snprintf(&status->class_devices[len], MAX_LEN - len,
470  ", %s", basename(p));
471  }
472 
473  found = 1;
474  }
475 
476  globfree(&paths);
477  return found;
478 }

480 static void get_port_info(const char *searchpath, int port,
481  struct switchtec_status *status)
482 {
483  int i;
484  char syspath[PATH_MAX];
485  glob_t paths;
486 
487  snprintf(syspath, sizeof(syspath), "%s/*:*:%02d.*/*:*:*/",
488  searchpath, port);
489 
490  glob(syspath, 0, NULL, &paths);
491 
492  for (i = 0; i < paths.gl_pathc; i++) {
493  char *p = paths.gl_pathv[i];
494 
495  snprintf(syspath, sizeof(syspath), "%s/vendor", p);
496  status->vendor_id = sysfs_read_int(syspath, 16);
497  if (status->vendor_id < 0)
498  continue;
499 
500  snprintf(syspath, sizeof(syspath), "%s/device", p);
501  status->device_id = sysfs_read_int(syspath, 16);
502  if (status->device_id < 0)
503  continue;
504 
505  if (get_class_devices(p, status)) {
506  if (status->pci_dev)
507  free(status->pci_dev);
508  status->pci_dev = strdup(basename(p));
509  }
510 
511  if (!status->pci_dev)
512  status->pci_dev = strdup(basename(p));
513  }
514 
515  globfree(&paths);
516 }

552  port++;
553  }
554 
555  return 0;

604 /*
605  * GAS map maps the hardware registers into user memory space.
606  * Needless to say, this can be very dangerous and should only
607  * be done if you know what you are doing. Any register accesses
608  * that use this will remain unsupported by Microsemi unless it's

611 gasptr_t switchtec_gas_map(struct switchtec_dev *dev, int writeable,
612  size_t *map_size)
613 {
614  int ret;
615  int fd;
616  void *map;
617  char respath[PATH_MAX];
618  struct stat stat;
619  struct switchtec_linux *ldev = to_switchtec_linux(dev);
620 
621  ret = dev_to_sysfs_path(ldev, "device/resource0_wc", respath,
622  sizeof(respath));
623 
624  if (ret) {
625  errno = ret;
626  return SWITCHTEC_MAP_FAILED;
627  }
628 
629  fd = open(respath, writeable ? O_RDWR : O_RDONLY);
630  if (fd < 0)
631  return SWITCHTEC_MAP_FAILED;
632 
633  ret = fstat(fd, &stat);
634  if (ret < 0)
635  return SWITCHTEC_MAP_FAILED;
636 
637  map = mmap(NULL, stat.st_size,
638  (writeable ? PROT_WRITE : 0) | PROT_READ,
639  MAP_SHARED, fd, 0);
640  close(fd);
641 
642  if (map == MAP_FAILED)
643  return SWITCHTEC_MAP_FAILED;
644 
645  if (map_size)
646  *map_size = stat.st_size;
647 
648  dev->gas_map = (gasptr_t __force)map;
649  dev->gas_map_size = stat.st_size;
650 
651  return (gasptr_t __force)map;
652 }
